
PAGINA DE GESTORES

SELECT T_GESTOR.COD, T_GESTOR.CPF, T_GESTOR.NOME, T_INSTITUICAO.SIGLA ,T_INSTITUICAO.NOME AS "INSTITUICAO_NOME"
FROM T_GESTOR, T_INSTITUICAO
WHERE T_GESTOR.COD_INSTITUICAO = T_INSTITUICAO.COD

PAGINA DE MULTAS

SELECT  T_MULTA.COD,T_MULTA.DATA_MULTA,T_MULTA.COD_GESTOR,(T_MULTA.NPARCELAS * T_PARCELA.VALOR) as "ValorTotal", T_MULTA.MOTIVOS
FROM T_MULTA, T_PARCELA, T_GESTOR
WHERE T_MULTA.COD_GESTOR = T_GESTOR.COD AND T_MULTA.COD = T_PARCELA.COD_MULTA
GROUP BY T_MULTA.COD,T_MULTA.DATA_MULTA,T_MULTA.COD_GESTOR,T_MULTA.MOTIVOS,(T_MULTA.NPARCELAS * T_PARCELA.VALOR)


PAGINA DE VERIFICACAO SE PAGOU OU NAO 

SELECT DISTINCT T_GESTOR.NOME, min(T_PARCELA.PAGO) as "Situação"
FROM T_GESTOR,T_PARCELA,T_MULTA
WHERE T_PARCELA.COD_MULTA = T_MULTA.COD AND T_MULTA.COD_GESTOR = T_GESTOR.COD
GROUP BY T_GESTOR.NOME
